00107 // Close a socket, preserving the current WinSock error code. When closesocket is called directly,
00108 // it will clear the WinSock error code (unless it fails). This function preserves the value of
00109 // the current error code when closesocket is called. It is intended to preserve the error
00110 // code when used as described by the following pseudo-code:
00111 // bool bindSocket(SOCKET sock)
00112 // {
00113 //   if (bind(sock) == SOCKET_ERROR())
00114 //   {
00115 //     // bind failed; don't reset the error code so the caller of bindSocket can obtain it
00116 //     closeSocketErr(sock);
00117 //     return false;
00118 //   }
00119 //   return true;
00120 // }
00121 inline int closeSocketErr(SOCKET sock)
00122 {
00123   int error = WSAGetLastError();
00124   int result = closesocket(sock);
00125   WSASetLastError(error);
00126   return result;
00127 }
